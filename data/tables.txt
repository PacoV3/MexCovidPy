CREATE TABLE UpdateDates (
	UpdateDateId INT PRIMARY KEY IDENTITY(1,1),
	DateUpdate DATE

	RegisterId VARCHAR(6) NOT NULL,
	UpdateDateId INT NOT NULL,
	Origin INT NOT NULL,
	Sector INT NOT NULL,
	EntityUm INT NOT NULL,
	Gender INT NOT NULL,
	BirthState INT NOT NULL,
	ResidenceState INT NOT NULL,
	ResidenceCity INT NOT NULL,
	PacientType INT NOT NULL,
	AdmissionDate DATE NOT NULL,
	SymptonsDate DATE NOT NULL,
	DeathDate DATE NOT NULL,
	Intubated INT NOT NULL,
	Pneumonia INT NOT NULL,
	Age INT NOT NULL,
	Nationality INT NOT NULL,
	Pregnancy INT NOT NULL,
	IndigenousLanguage INT NOT NULL,
	Indigenous int,
	Diabetes INT NOT NULL,
	Epoc INT NOT NULL,
	Asthma INT NOT NULL,
	Immunosuppression INT NOT NULL,
	Hypertension INT NOT NULL,
	OtherCom INT NOT NULL,
	Cardiovascular INT NOT NULL,
	Obesity INT NOT NULL,
	ChronicKidney INT NOT NULL,
	Smoking INT NOT NULL,
	OtherCase INT NOT NULL,
	TakeSample int,
	Result INT NOT NULL,
	FinalClassification int,
	Migrant INT NOT NULL,
	NationalityCountry VARCHAR(70) NOT NULL,
	OriginCountry VARCHAR(70) NOT NULL,
	Uci INT NOT NULL
)

CREATE TABLE DiaryResume(
	DateId INT PRIMARY KEY,
	Confirmed INT,
	Negatives INT,
	Suspects INT,
	Deaths INT,
	Recovered INT,
	Active INT,
)

CREATE TABLE PatientSummary (
	RegisterId VARCHAR(6) PRIMARY KEY,
	DeathPlace BIT,
	DaysUntilAdmission INT,
	DaysUntilDeath INT,
	DaysIntubated INT
)

CREATE TABLE States(
	StateId INT PRIMARY KEY,
	StateName NVARCHAR(50),
	Abbreviation NVARCHAR(5),
	Latitude FLOAT,
	Longitude FLOAT
)

CREATE TABLE Cities(
	CityId INT NOT NULL,
	StateId INT NOT NULL,
	CityName NVARCHAR(80),
	Latitude FLOAT,
	Longitude FLOAT
)

CREATE TABLE DiaryResumeStates(
	DateId INT NOT NULL,
	StateId INT NOT NULL,
	Confirmed INT,
	Negatives INT,
	Suspects INT,
	Deaths INT,
	Recovered INT,
	Active INT,
)

CREATE TABLE DiaryResumeCities(
	DateId INT NOT NULL,
	StateId INT NOT NULL,
	CityId INT NOT NULL,
	Confirmed INT,
	Negatives INT,
	Suspects INT,
	Deaths INT,
	Recovered INT,
	Active INT,
)
